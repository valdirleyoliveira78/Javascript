$(document).read(function () {
    $('#frmCadastro').validate({
        rules: {
            nome: {
                required: true,
                minlenght: 4,
            }
        },

        email: {
            required: true,
            email: true,
        },


        senha: {
            required: true,
            minlenght: 8
        },

        senhaConfirmacao: {
            required: true,
            equalTo: senha

        },

        termos: {
            required: true,

        },

        messages: {
            nome: {
                required: "Favor Digitar o Nome",
                minlenght: "O Campo Nome deve ter no mínimo 4 caracteres"
            },  
           email: {
                required: "Favor Digitar o Nome",
                email: "Email Inválido"
            },   
           senha: {
                required: "Campo Obrigat[orio",
            },
            senhaConfirmacao:{
                required: "Campo Obrigat[orio",
                equalTo: "As senhas digitadas não são iguais",
            },
            termos:{
                required: "Para se cadastrar é necessários aceitar os termos",
                },
            
        }
    })

})
